// Check if Nexus is online
async function isNexusOnline() {
  const response = await fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/health');
  const health = await response.json();
  return health.status === 'operational';
}

Display Live Stats Widget

// Update website with live stats
async function updateStats() {
  const stats = await fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/stats')
    .then(r => r.json());
  
  document.getElementById('servers').textContent = stats.serverCount;
  document.getElementById('users').textContent = stats.userCount.toLocaleString();
  document.getElementById('uptime').textContent = stats.uptime + ' min';
}

// Update every 30 seconds
setInterval(updateStats, 30000);
updateStats();
