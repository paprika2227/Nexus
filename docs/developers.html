üìà Track Server Growth

import requests
import matplotlib.pyplot as plt

# Get and visualize growth data
def visualize_growth():
    response = requests.get('YOUR_API_URL/api/v1/growth')
    growth = response.json()
    
    # Extract dates and values
    dates = [entry['date'] for entry in growth['joins']]
    joins = [entry['joins'] for entry in growth['joins']]
    
    # Create chart
    plt.plot(dates, joins, label='Joins', color='green')
    plt.xlabel('Date')
    plt.ylabel('Servers')
    plt.title('Nexus Growth Trend')
    plt.legend()
    plt.show()

visualize_growth()
