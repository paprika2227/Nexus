1. Get Bot Stats (No Auth Required)

// JavaScript
const response = await fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/stats');
const data = await response.json();
console.log(`Nexus is protecting ${data.serverCount} servers!`);

// Python
import requests
r = requests.get('https://regular-puma-clearly.ngrok-free.app/api/v1/stats')
print(f"Servers: {r.json()['serverCount']}")

// curl
curl https://regular-puma-clearly.ngrok-free.app/api/v1/stats

2. Check System Health

// Get real-time health status
fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/health')
  .then(r => r.json())
  .then(data => {
    console.log(`Bot Status: ${data.status}`);
    console.log(`Servers: ${data.bot.servers}`);
    console.log(`Ping: ${data.bot.ping}ms`);
  });

3. View Growth Analytics

// Get 30-day growth data
fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/growth')
  .then(r => r.json())
  .then(data => {
    console.log(`Net Growth: ${data.netGrowth} servers`);
    console.log(`Joins per day:`, data.joins);
  });

4. Get Command Analytics

// See most popular commands
fetch('https://regular-puma-clearly.ngrok-free.app/api/v1/command-analytics')
  .then(r => r.json())
  .then(data => {
    console.log('Top Commands:', data.topCommands);
  });
